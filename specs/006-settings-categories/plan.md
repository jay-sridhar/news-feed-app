# Implementation Plan: Settings Page with Category Personalisation

**Branch**: `006-settings-categories` | **Date**: 2026-03-02 | **Spec**: `specs/006-settings-categories/spec.md`
**Input**: Feature specification from `/specs/006-settings-categories/spec.md`

## Summary

Add a Settings screen accessible from a gear icon in the app header. Users can toggle which
of the 5 news category tabs are visible. At least one category must remain enabled. The
active tab auto-switches if its category is deselected. The dark mode toggle moves into
Settings. Preferences persist to localStorage. All state is managed by extending
`CategoryContext` — no new library, no new context provider.

## Technical Context

**Language/Version**: TypeScript 5.x (strict mode)
**Primary Dependencies**: React 18, Vite 5, Tailwind CSS 3 — no new dependencies
**Storage**: localStorage (`newsflow_enabled_categories: CategoryId[]`)
**Testing**: Playwright (`@playwright/test`) — Chromium, 390×844 viewport
**Target Platform**: SPA, Vercel free tier (static)
**Project Type**: Web application (client-side SPA)
**Performance Goals**: Settings open/close in < 100 ms (pure state toggle, no fetch)
**Constraints**: No external state libraries; localStorage only (no network for prefs); min-one guard always enforced
**Scale/Scope**: 5 categories, 1 localStorage key, 3 new files, ~4 modified files

### Key Design Details

#### CategoryContext changes (`src/context/CategoryContext.tsx`)

```ts
// New state
const [enabledCategories, setEnabledCategories] = useState<CategoryId[]>(() => {
  try {
    const stored = localStorage.getItem('newsflow_enabled_categories')
    if (stored) {
      const parsed = JSON.parse(stored) as unknown
      if (Array.isArray(parsed)) {
        const valid = parsed.filter((id): id is CategoryId => id in CATEGORY_MAP)
        if (valid.length > 0) return valid
      }
    }
  } catch { /* ignore */ }
  return CATEGORIES.map(c => c.id)  // default: all enabled
})
const [isSettingsOpen, setIsSettingsOpen] = useState(false)

function toggleCategory(id: CategoryId): void {
  setEnabledCategories(prev => {
    const isEnabled = prev.includes(id)
    if (isEnabled && prev.length === 1) return prev  // guard: min-one
    const next = isEnabled ? prev.filter(x => x !== id) : [...prev, id]
    localStorage.setItem('newsflow_enabled_categories', JSON.stringify(next))
    return next
  })
}

function openSettings(): void { setIsSettingsOpen(true) }

function closeSettings(): void {
  setIsSettingsOpen(false)
  // Auto-switch if active tab was just deselected
  setActiveCategory(prev =>
    enabledCategories.includes(prev as CategoryId) || prev === 'bookmarks'
      ? prev
      : enabledCategories[0]
  )
}
```
